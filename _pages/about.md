// Fetch Google Scholar citations
(function() {
  const badge = document.getElementById('citation-badge');
  if (!badge) return;
  
  // Try to fetch from Google Scholar via a proxy
  fetch('https://api.allorigins.win/get?url=' + encodeURIComponent('https://scholar.google.com/citations?user=-cNWmJMAAAAJ'))
    .then(response => response.json())
    .then(data => {
      const html = data.contents;
      // Extract citation count from the HTML
      const match = html.match(/Citations[^\d]*(\d[\d,]*)/i);
      if (match) {
        const citations = match[1].replace(/,/g, '');
        const formattedCitations = citations > 1000 ? (citations / 1000).toFixed(1) + 'k' : citations;
        badge.src = 'https://img.shields.io/badge/Citations-' + formattedCitations + '-2E5077?style=for-the-badge&logo=google-scholar&logoColor=white&labelColor=1A2A3A';
      } else {
        badge.src = 'https://img.shields.io/badge/Citations-1.8k-2E5077?style=for-the-badge&logo=google-scholar&logoColor=white&labelColor=1A2A3A';
      }
    })
    .catch(() => {
      // Fallback to static value
      badge.src = 'https://img.shields.io/badge/Citations-1.8k-2E5077?style=for-the-badge&logo=google-scholar&logoColor=white&labelColor=1A2A3A';
    });
})();

  const pvKey = 'knightzjz_page_views';
  const uvKey = 'knightzjz_unique_visitors';
  const visitorId = localStorage.getItem('knightzjz_visitor_id');
  
  // Page Views
  let pvCount = localStorage.getItem(pvKey);
  pvCount = pvCount ? parseInt(pvCount) + 1 : 1;
  localStorage.setItem(pvKey, pvCount);
  document.getElementById('pv-count').textContent = pvCount.toLocaleString();
  
  // Unique Visitors (simple approach - per browser)
  let uvCount = localStorage.getItem(uvKey);
  uvCount = uvCount ? parseInt(uvCount) : 0;
  
  if (!visitorId) {
    uvCount += 1;
    localStorage.setItem(uvKey, uvCount);
    localStorage.setItem('knightzjz_visitor_id', Date.now().toString());
  }
  document.getElementById('uv-count').textContent = uvCount.toLocaleString();
})();
